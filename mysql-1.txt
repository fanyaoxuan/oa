约束：
	约束的信息通过表user_constraints查询
	也可以通过user_cons_columns查看详细信息

非空约束（not null）：
	create tabel member(
		mid number,
		name varchar2(20) not null
	) 
唯一约束(unique,nk):
	create table member(
		mid number,
		email varchar2(20) unique
	) 
	
	空值并不受唯一约束的限定。
	create table member(
		mid number,
		email varchar2(20),
		constraint 约束名 unique(列名)
	) 
	
主键约束(primary key,pk)
	一张表一个主键
	create table member(
		mid number,
		email varchar2(20),
		constraint 约束名 primary key(列名)
	) 
	一张表多个主键（复合主键）
	create table member(
		mid number,
		email varchar2(20),
		constraint 约束名 primary key(列名，列名)
	) 
检查约束（check,ck）
	create tabel member(
		mid number,
		age number(3),
		constraint 约束名 check(age between 0 and 250)
	)

外键约束（foreign key，fk）
	create tabel member(
		mid number,
		age number(3),
		deptno number,
		constraint 约束名 foreign key(deptno) refrences dept(deptno)
	)


	强制删除存在外键关系的表：
	drop table 表名 cascade constraint;

	数据的级联操作：
	create tabel member(
		mid number,
		age number(3),
		deptno number,
		constraint 约束名 foreign key(deptno) 
		refrences dept(deptno) on delete cascade;
	)

	级联更新：
	create tabel member(
		mid number,
		age number(3),
		deptno number,
		constraint 约束名 foreign key(deptno) 
		refrences dept(deptno) on delete set null;
	)
